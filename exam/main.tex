    \question[10] 下列哪个算式是正确的？(\hspace*{2em})

    \begin{choices}
        \choice $3 \times (4 + 5) = 27$
        \CorrectChoice $3 \times (4 + 5) = 3 \times 4 + 3 \times 5$
        \choice $3 \times (4 + 5) = 4 \times (3 + 5)$
        \choice $3 \times (4 + 5) = (3 + 4) \times (3 + 5)$
    \end{choices}
